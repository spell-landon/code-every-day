function find_average(array) {
  // Solution
  if (array.length === 0) return 0;
  let sum = 0;
  array.forEach((num) => {
    sum += num;
  });
  return sum / array.length;
}

Python #2. Invert Values

> This one wanted to invert a list of numbers

```python
def invert(lst):
    list = []
    for i in lst:
        if i > 0:
            list.append(-abs(i))
        elif i < 0:
            list.append(abs(i))
        else:
            list.append(0)
    return list
```
